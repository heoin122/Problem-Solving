#include <iostream>
#include <string.h>
#include <list>

using namespace std;

char arr[1000010];

int main() {
	int testcase;
	scanf("%d", &testcase);
	while(testcase--) {
		scanf("%s", arr);
		int n = strlen(arr);
		list<char> l;
		list<char>::iterator cursor = l.end();
		for(int i = 0; i<n; i++){
			if(arr[i] == '<') {
				if(cursor != l.begin()) cursor--;
			} else if(arr[i] == '>') {
				if(cursor != l.end()) cursor++;
			} else if(arr[i] == '-') {
				if(cursor == l.begin()) continue;
				cursor--;
				cursor = l.erase(cursor);
			} else {
				l.insert(cursor, arr[i]);
			}
		}
		for(auto it = l.begin(); it != l.end(); it++){
			printf("%c", *it);
		} printf("\n");
	}
	return 0;
}
