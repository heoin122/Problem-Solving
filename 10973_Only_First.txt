/*
	구현문제였다.
	Only First
*/
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int a[10010];
int main() {
	int n;
	scanf("%d", &n);
	for(int i = 0; i<n; i++) scanf("%d", &a[i]);
	
	bool ans = true;
	for(int i = 0; i<n; i++){
		if(a[i] != i + 1) {
			ans = false;
			break;
		}
	}
	if(ans) {
		printf("-1");
		exit(0);
	}
	
	if(a[n-2] > a[n-1]) {
		for(int i = 0; i<n-2; i++) printf("%d ", a[i]);
		printf("%d %d", a[n-1], a[n-2]);
	} else {
		vector <int> ret;
		ret.push_back(a[n-1]);
		ret.push_back(a[n-2]);
				
		for(int i = n-2; i>=0; i--){
			if(a[i-1] < a[i]) {
				ret.push_back(a[i-1]);	
			} else {
				int MAX = 0;
				for(auto it = ret.begin(); it != ret.end(); it++) {
					if(*it < a[i-1] && MAX < *it) {
						MAX = *it;
					}
				}
				ret.push_back(a[i-1]);
				sort(ret.begin(), ret.end());
				
				for(int j = 0; j<i-1; j++) printf("%d ", a[j]);
				printf("%d ", MAX);
				
				for(int j = ret.size()-1; j >= 0; j--) {
					if(ret[j] != MAX) {
						printf("%d ", ret[j]);
					}
				}
				break;
			}
			
		}
	}
	return 0;
}
