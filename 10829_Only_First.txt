/*
10829_이진수 변환

	단순 구현 문제였다.
	Only_First
*/
#include <iostream>
#include <vector>
using namespace std;

int main() {
	long long int n;
	scanf("%lld", &n);
	vector <int> ans;
	while(n) {
		ans.push_back(n % 2);
		n /= 2;
	}
	for(int i = ans.size()-1; i>=0; i--) printf("%d", ans[i]);
	return 0;
}
