/*
	단순 수학 구현문제이다.
	Only_First
*/

#include <iostream>
#include <cmath>

using namespace std;

int main() {
	int n, h, w;
	scanf("%d %d %d", &n, &h, &w);
	double cut = h*h + w*w;
	cut = pow(cut, 0.5);
	
	int temp;
	for(int i = 0; i<n; i++){
		scanf("%d", &temp);
		if(temp <= cut) printf("DA\n");
		else printf("NE\n");
	}
	return 0;
}
