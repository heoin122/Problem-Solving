/*
	대소 비교 구현문제였다.
	Only First
*/

#include <iostream>
using namespace std;

int main() {
	int a, b, c;
	while(true){
		scanf("%d %d %d", &a, &b, &c);
		if(a == 0 && b == 0 && c == 0) break;
		
		if(a == 0 || b == 0 || c == 0) {
				printf("Invalid\n");	
				continue;
		}
		if(a > b && a > c) {
			if(a >= b + c){
				printf("Invalid\n");	
				continue;
			}
		}
		else if(b > c) {
			if(b >= a + c) {
				printf("Invalid\n");	
				continue;
			}
		} else {
			if(c >= a + b){
				printf("Invalid\n");	
				continue;
			}
		}
		if(a == b && b == c) printf("Equilateral\n");
		else if(a == b || b == c || a == c) printf("Isosceles\n");
		else printf("Scalene\n");
	}
	return 0;
}
