/*
11660_구간 합 구하기5

	2차원 부분합을 이용하여 해결하면 되는 문제이다.
	Only First

*/
#include <iostream>
using namespace std;

const int MAXN = 1030;

int a[MAXN][MAXN], sum[MAXN][MAXN];

int main() {
	int n, m;
	scanf("%d %d", &n, &m);
	for(int i = 1; i<=n; i++){
		for(int j = 1; j<=n; j++){
			scanf("%d ", &a[i][j]);
		}
	}
	for(int i = 1; i<=n; i++){
		for(int j = 1; j<=n; j++){
			sum[i][j] = a[i][j] + sum[i-1][j] + sum[i][j-1] - sum[i-1][j-1];
		}
	}	
	int x1, x2, y1, y2;
	for(int i = 1; i<=m; i++){
		scanf("%d %d %d %d", &x1, &y1, &x2, &y2);
		printf("%d\n", sum[x2][y2] - sum[x1-1][y2] - sum[x2][y1-1] + sum[x1-1][y1-1]);
		
	}
	return 0;
}
