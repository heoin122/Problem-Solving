	3986_좋은 단어 

	문제의 설명이 어렵지만, 괄호 쌍 맞추기처럼 
	스택을 이용하여 푸는 문제이다.
	Only First
*/
#include <iostream>
#include <string.h>
#include <string>
#include <stack>

using namespace std;

char arr[100010];
int main() {
	int testcase;
	int ans = 0;

	scanf("%d", &testcase);

	while(testcase--) {
		scanf("%s", arr);
		int n = strlen(arr);
		stack <char> st;
		for(int i = 0; i<n; i++){
			if(st.empty()) st.push(arr[i]);
			else {
				char top = st.top();
				if(top == arr[i]) {
					st.pop();
				} else {
					st.push(arr[i]);
				}
			}
		}
		if(st.empty()) ++ans;
	}
	printf("%d", ans);
	return 0;
}
