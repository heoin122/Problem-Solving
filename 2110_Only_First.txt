/*
	간단한 이분법 문제였다.
	Only First
*/

#include <iostream>
#include <algorithm>

using namespace std;

int a[200010];
int n, m;

bool isPossible(int dis) {
	int ret = 1;
	int num = a[0];
	for(int i = 1; i<n; i++){
		if(a[i] - num >= dis) {
			ret++;
			num = a[i];
		}
	}	
	return ret >= m;
}
int main() {
	scanf("%d %d", &n, &m);
	for(int i = 0; i<n; i++){
		scanf("%d", &a[i]);
	}
	sort(a, a+n);
	int left = 1;
	int right = 1000000000;
	
	while(left <= right) {
		int mid = (left + right) / 2;
		if(isPossible(mid)) {
			left = mid + 1;
		} else {
			right = mid - 1;
		}
	}
	printf("%d", right);
	return 0;
}
