/*
	정보올림피아드 색종이와 별 다를게 없다.
	구현 문제.
*/
#include <iostream>
using namespace std;

int a[110][110], cnt[110];
int main() {
	int n;
	scanf("%d", &n);
	
	int boX, boY, upX, upY;
	for(int t = 1; t<=n; t++){
		scanf("%d %d %d %d", &boX, &boY, &upX, &upY);
		for(int i = boX; i<boX + upX; i++){
			for(int j = boY; j < boY + upY; j++){
				if(a[i][j] != 0) {
					cnt[a[i][j]]--;
				}
				a[i][j] = t;
				cnt[t]++;
			}
		}
	}
	for(int i = 1; i<=n; i++) printf("%d\n", cnt[i]);
	return 0;
}
