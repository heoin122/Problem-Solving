	단순 구현 문제였다.
	Only_First
*/
#include <iostream>
#include <cstring>
using namespace std;

char arr[10010];
int main() {
	long long int sum = 0;
	int p = 0;
	while(scanf("%s", arr) != EOF) {
		int size = strlen(arr);
		for(int i = 0; i<size; i++) {
			if(arr[i] == ',') {
				sum += p;
				p = 0;
			} else {
				p *= 10;
				p += arr[i] - '0';
			}
		}	
	}
	sum += p;
	printf("%lld", sum);
	return 0;
}
