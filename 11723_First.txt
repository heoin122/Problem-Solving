#include <iostream>
#include <string.h>
using namespace std;

const int FULL = (1<<21) - 1;
int main() {
	char type[10];
	int n, S = 0, tv;
	scanf("%d", &n);
	for(int i = 1; i<=n; i++){
		scanf("%s %d", type, &tv);
		if(!strcmp(type, "add")) {
			S |= (1<<tv);		
		} else if(!strcmp(type, "remove")) {
			S &= ~(1<<tv);
		} else if(!strcmp(type, "check")) {
			if((S & (1<<tv)) > 0) printf("1\n");
			else printf("0\n");
		} else if(!strcmp(type, "toggle")) {
			S ^= (1<<tv);
		} else if(!strcmp(type, "all")) {
			S = FULL;
		} else if(!strcmp(type, "empty")) {
			S = 0;
		}
	}
	return 0;
}
