#include <iostream>
#include <string>
#include <string.h>
using namespace std;

int main() {
	char S[110];
	string CODE = "Simon says";
	int n;
	scanf("%d\n", &n);
	for(int i = 0; i<n; i++){
		cin.getline(S, 110);
		string temp = "";
		for(int j = 0; j<10; j++) temp += S[j];
		
		if(temp == CODE) {
			int size = strlen(S);
			for(int i = 10; i< size; i++){
				printf("%c", S[i]);
			}
			printf("\n");
		}
	}
	return 0;
}
